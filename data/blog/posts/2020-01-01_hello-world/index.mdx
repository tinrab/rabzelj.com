pub fn fft(input: &[Complex64]) -> Vec<Complex64> {
    let n = input.len();
    if n <= 1 {
        return input.to_vec();
    }
    if !n.is_power_of_two() {
        panic!("Input length must be a power of 2");
    }
    let mut even: Vec<Complex64> = input.iter().step_by(2).copied().collect();
    let mut odd: Vec<Complex64> = input.iter().skip(1).step_by(2).copied().collect();
    even = fft(&even);
    odd = fft(&odd);
    let mut result = vec![Complex64::new(0.0, 0.0); n];
    for k in 0..n/2 {
        let t = (-2.0 * PI * k as f64 / n as f64).exp();
        let t = Complex64::new(t.cos(), t.sin());
        result[k] = even[k] + t * odd[k];
        result[k + n/2] = even[k] - t * odd[k];
    }
    result
}
